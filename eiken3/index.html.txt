    function startMode(mode) {
        currentSessionResults = [];
        currentScore = 0;
        queueIndex = 0;
        document.getElementById('score-display').innerText = `SCORE: ${currentScore}`;
        document.getElementById('app-nav').style.display = 'flex';

        if (mode === 'random') {
            currentQueue = fullMasterData.map(item => item.id);
            currentQueue.sort(() => Math.random() - 0.5);
        } else if (mode === 'review') {
            currentQueue = fullMasterData.filter(item => item.lastStatus === "不正解").map(item => item.id);
            currentQueue.sort(() => Math.random() - 0.5);
        }

        if (currentQueue.length > 0) {
            pickRandomIdol();
            showScreen('quiz-screen');
            loadQuizQuestion();
        }
    }

    function startSingleQuestion(id) {
        currentSessionResults = [];
        currentScore = 0;
        queueIndex = 0;
        document.getElementById('score-display').innerText = `SCORE: ${currentScore}`;
        document.getElementById('app-nav').style.display = 'flex';
        
        currentQueue = [id];
        pickRandomIdol();
        showScreen('quiz-screen');
        loadQuizQuestion();
    }

    function pickRandomIdol() {
        const randomIndex = Math.floor(Math.random() * idolSupportDatabase.length);
        currentIdol = idolSupportDatabase[randomIndex];
    }

    function loadQuizQuestion() {
        isAnswered = false;
        const activeId = currentQueue[queueIndex];
        const questionData = fullMasterData.find(item => item.id === activeId);

        document.getElementById('quiz-theme').innerText = `第${questionData.id}問: ${questionData.meaning}`;
        document.getElementById('quiz-question').innerHTML = questionData.sentence.replace("______", `<span class="blank">[ ??? ]</span>`);
        
        const optionsBox = document.getElementById('quiz-options');
        optionsBox.innerHTML = '';

        questionData.options.forEach(opt => {
            const btn = document.createElement('button');
            btn.className = 'option-btn';
            btn.innerText = opt;
            btn.onclick = () => checkQuizAnswer(opt, btn, questionData);
            optionsBox.appendChild(btn);
        });

        document.getElementById('idol-cheer-box').style.display = 'none';
        document.getElementById('quiz-feedback').style.display = 'none';
        document.getElementById('quiz-next-btn').style.display = 'none';
    }

    function checkQuizAnswer(selected, buttonElement, data) {
        if (isAnswered) return;
        isAnswered = true;

        const optionBtns = document.querySelectorAll('.option-btn');
        optionBtns.forEach(btn => btn.style.cursor = 'default');

        const feedbackBox = document.getElementById('quiz-feedback');
        const feedbackDetail = document.getElementById('quiz-feedback-detail');
        const cheerBox = document.getElementById('idol-cheer-box');
        const cheerGroup = document.getElementById('idol-group');
        const cheerName = document.getElementById('idol-name');
        const cheerText = document.getElementById('idol-bubble-text');

        let isCorrect = (selected === data.answer);
        data.timesPlayed += 1;

        cheerGroup.innerText = currentIdol.group;
        cheerName.innerText = currentIdol.name;
        cheerName.style.color = currentIdol.color;
        cheerBox.querySelector('.idol-bubble').style.borderColor = currentIdol.color;

        if (isCorrect) {
            data.correctCount += 1;
            data.lastStatus = "正解";
            currentScore += 10;
            document.getElementById('score-display').innerText = `SCORE: ${currentScore}`;
            buttonElement.style.backgroundColor = '#2ECC71';
            buttonElement.style.color = 'white';
            buttonElement.style.borderColor = '#2ECC71';
            cheerText.innerText = currentIdol.correct;
        } else {
            data.incorrectCount += 1;
            data.lastStatus = "不正解";
            buttonElement.style.backgroundColor = '#E74C3C';
            buttonElement.style.color = 'white';
            buttonElement.style.borderColor = '#E74C3C';
            
            optionBtns.forEach(btn => {
                if (btn.innerText === data.answer) {
                    btn.style.borderColor = '#2ECC71';
                    btn.style.backgroundColor = '#E8F8F5';
                }
            });
            cheerText.innerText = currentIdol.incorrect;
        }

        document.getElementById('quiz-question').innerHTML = data.sentence.replace("______", `<span class="blank">${data.answer}</span>`);

        feedbackDetail.innerHTML = `
            <strong>【覚える重要語句】</strong> ${data.expression}<br>
            <strong>【日本語訳】</strong> ${data.translation}<br><br>
            <strong>💡解説：</strong><br>${data.explanation}
        `;

        cheerBox.style.display = 'block';
        feedbackBox.style.display = 'block';
        document.getElementById('quiz-next-btn').style.display = 'block';

        currentSessionResults.push({
            id: data.id,
            meaning: data.meaning,
            expression: data.expression,
            fullSentence: data.sentence.replace("______", `<u><strong>${data.answer}</strong></u>`),
            translation: data.translation,
            explanation: data.explanation,
            isCorrect: isCorrect
        });
    }

    function handleNext() {
        queueIndex++;
        if (queueIndex < currentQueue.length) {
            pickRandomIdol();
            loadQuizQuestion();
        } else {
            showResultScreen();
        }
    }

    function showResultScreen() {
        showScreen('result-screen');
        document.getElementById('result-title').innerHTML = `全問終了！獲得スコア: <strong>${currentScore} 点</strong>`;
        
        const reviewListContainer = document.getElementById('result-review-list');
        reviewListContainer.innerHTML = '';

        currentSessionResults.forEach((res) => {
            const item = document.createElement('div');
            item.className = `review-item ${res.isCorrect ? 'correct' : 'incorrect'}`;
            item.innerHTML = `
                <div class="review-header" onclick="toggleAccordion(this)">
                    <span>${res.isCorrect ? '〇' : '×'} 第${res.id}問: ${res.meaning}</span>
                    <span>▼</span>
                </div>
                <div class="review-body">
                    <strong>【重要項目】</strong> ${res.expression}<br>
                    <strong>【英文】</strong> ${res.fullSentence}<br><br>
                    <strong>【訳】</strong> ${res.translation}<br><br>
                    <strong>【解説】</strong> ${res.explanation}
                </div>
            `;
            reviewListContainer.appendChild(item);
        });
    }

    function toggleAccordion(header) {
        const body = header.nextElementSibling;
        const isVisible = body.style.display === 'block';
        body.style.display = isVisible ? 'none' : 'block';
        header.querySelector('span:last-child').innerText = isVisible ? '▼' : '▲';
    }

    function goToTop() {
        renderTopScreen();
    }

    function showScreen(screenId) {
        const screens = ['top-screen', 'quiz-screen', 'result-screen'];
        screens.forEach(id => {
            document.getElementById(id).classList.remove('active');
        });
        document.getElementById(screenId).classList.add('active');
    }
